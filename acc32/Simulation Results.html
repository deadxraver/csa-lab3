loop:
    load         i                           ; acc = i
    beqz         end_loop                    ; i == 0 ? -> end_loop
    sub          one                         ; i--
    store        i                           ; save i
    load         n2                          ; load n2

    store        n2                          ; save n2
    load         n1                          ; acc = n1

    store        n1
    jmp          loop                        ; next iteration
